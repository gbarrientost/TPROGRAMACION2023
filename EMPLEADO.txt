CREATE TABLE TIPO_EMPLEADO
(TIPO_EMPLEADO INT(2) PRIMARY KEY AUTO_INCREMENT,
DESC_TIPO_EMPLEADO VARCHAR(25) NOT NULL
);


CREATE TABLE EMPLEADO
(ID_EMPLEADO INT(5),
NUMRUT INT(11) NOT NULL,
DVRUT VARCHAR(1) NOT NULL,
PRIMER_NOMBRE VARCHAR(20) NOT NULL,
SNOMBRE VARCHAR(20),
APPATERNO VARCHAR(20) NOT NULL,
APMATERNO VARCHAR(20) NOT NULL,
FECHA_CONTRATO DATE NOT NULL,
TIPO_EMPLEADO INT(2) NOT NULL,
PRIMARY KEY (ID_EMPLEADO),
FOREIGN KEY (TIPO_EMPLEADO) REFERENCES TIPO_EMPLEADO (TIPO_EMPLEADO));

CREATE TABLE VENTAS
(NRO_BOLETA INT(11) PRIMARY KEY,
ID_EMPLEADO INT(5) NOT NULL,
FECHA_BOLETA DATE NOT NULL,
MONTO_BOLETA INT(8) NOT NULL,
FOREIGN KEY (ID_EMPLEADO) REFERENCES EMPLEADO(ID_EMPLEADO)
);

CREATE TABLE COMISION_VENTAS
(NRO_BOLETAS INT(11) PRIMARY KEY,
MONTO_COMISION INT(8) NOT NULL,
FOREIGN KEY (NRO_BOLETAS) REFERENCES VENTAS(NRO_BOLETA)
);
